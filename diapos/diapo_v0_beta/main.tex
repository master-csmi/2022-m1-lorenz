	\begin{frame}[allowframebreaks]{Lorenz system}
		
		The system :
		$$\left\{\begin{aligned} 
			x'&=\sigma(y(t)-x(t)) \\
			y'&=x(t)(r-z(t))-y(t) \\
			z'&=x(t)y(t)-bz(t)
		\end{aligned}\right.$$
	
		where
		
		\begin{enumerate}[\textbullet]
			\item $\sigma > 0$  relates to the Prandtl number
			\item $r > 0$  relates to the Rayleigh
			\item $b > 0$ is a geometric factor
		\end{enumerate}
		
	\end{frame}
